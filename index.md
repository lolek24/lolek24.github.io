<script>
    function showTab(tabName) {
        var panels = document.querySelectorAll('.sap-tab-panel');
        panels.forEach(function(panel) { panel.classList.remove('sap-active'); });
        document.getElementById(tabName).classList.add('sap-active');
        var filters = document.querySelectorAll('.sap-tab-filter');
        filters.forEach(function(filter) { filter.classList.remove('sap-selected'); });
        event.currentTarget.classList.add('sap-selected');
    }
</script>
